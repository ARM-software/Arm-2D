175/*!
176 * \brief initialize/implement a given film (arm_2d_helper_file_t) object
177 * at compile-time.
178 * \param[in] __sprites_tile the sprites tile
179 * \param[in] __width the width of each frame
180 * \param[in] __height the height of each frame
181 * \param[in] __column the number of frames per row in the sprite tile
182 * \param[in] __frame_count the total number of frames in the sprite tile
183 * \param[in] __period the period per-frame
184 * \note __period is used as a reference for applications. The helper service
185 * doesn't use it at all.
186 */
187#define impl_film( __sprites_tile, \
188 __width, \
189 __height, \
190 __column, \
191 __frame_count, \
192 __period) \
193 { \
194 .use_as__arm_2d_tile_t = \
195 impl_child_tile((__sprites_tile), 0, 0, (__width), (__height)), \
196 .hwColumn = (__column), \
197 .hwFrameNum = (__frame_count), \
198 .hwPeriodPerFrame = (__period), \
199 }
