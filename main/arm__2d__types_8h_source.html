187/*!
188 * \brief the colour type for brga8888
189 *
190 * \details In most cases four equal-sized pieces of adjacent memory are used,
191 * one for each channel, and a 0 in a channel indicates black color or
192 * transparent alpha, while all-1 bits indicates white or fully opaque
193 * alpha. By far the most common format is to store 8 bits (one byte)
194 * for each channel, which is 32 bits for each pixel.
195 *
196 * (source: https://en.wikipedia.org/wiki/RGBA_color_model#ARGB32)
197 */
198typedef union arm_2d_color_bgra8888_t {
199 uint32_t tValue;
200 struct {
201 uint32_t u8B : 8;
202 uint32_t u8G : 8;
203 uint32_t u8R : 8;
204 uint32_t u8A : 8;
205 };
206} arm_2d_color_bgra8888_t;
207
208/*!
209 * \brief the colour type for rgb888 (compliant with ccca888 and bgra8888)
210 *
211 * \details In most cases four equal-sized pieces of adjacent memory are used,
212 * one for each channel, and a 0 in a channel indicates black color or
213 * transparent alpha, while all-1 bits indicates white or fully opaque
214 * alpha. By far the most common format is to store 8 bits (one byte)
215 * for each channel, which is 32 bits for each pixel.
216 *
217 * (source: https://en.wikipedia.org/wiki/RGBA_color_model#ARGB32)
218 */
219typedef union arm_2d_color_rgb888_t {
220 uint32_t tValue;
221 struct {
222 uint32_t u8B : 8;
223 uint32_t u8G : 8;
224 uint32_t u8R : 8;
225 uint32_t : 8;
226 };
227} arm_2d_color_rgb888_t;

229/*!
230 * \brief the colour type for any 32bit colour formats which has an alpha channel on its 3rd byte.
231 *
232 * \details In most cases four equal-sized pieces of adjacent memory are used,
233 * one for each channel, and a 0 in a channel indicates black color or
234 * transparent alpha, while all-1 bits indicates white or fully opaque
235 * alpha. By far the most common format is to store 8 bits (one byte)
236 * for each channel, which is 32 bits for each pixel.

244 uint8_t u8A;
245 };
246} arm_2d_color_ccca8888_t;
