122/*!
123 * \brief set an alarm with given period and check the status
124 *
125 * \param[in] wMS a time period in millisecond
126 *
127 * \return bool whether it is timeout
128 */
129__STATIC_INLINE bool arm_2d_helper_is_time_out(uint32_t wMS)
130{
131 int64_t lTimestamp = arm_2d_helper_get_system_timestamp();
132 static int64_t s_lTimestamp = 0;
133 if (0 == s_lTimestamp) {
134 s_lTimestamp = arm_2d_helper_convert_ms_to_ticks(wMS);
135 s_lTimestamp += lTimestamp;
136
137 return false;
138 }
139
140 if (lTimestamp >= s_lTimestamp) {
141 s_lTimestamp = arm_2d_helper_convert_ms_to_ticks(wMS) + lTimestamp;
142 return true;
143 }
144
145 return false;
146}
