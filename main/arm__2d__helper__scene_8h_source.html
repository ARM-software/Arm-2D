58/*!
59 * \brief scene switching mode
60 */
61typedef enum {
62
63 /* valid switching visual effects begin */
64 ARM_2D_SCENE_SWITCH_MODE_NONE = 0, //!< no switching visual effect
65 ARM_2D_SCENE_SWITCH_MODE_USER = 1, //!< user defined switching visual effect
66 ARM_2D_SCENE_SWITCH_MODE_FADE_WHITE = 2, //!< fade in fade out (white)
67 ARM_2D_SCENE_SWITCH_MODE_FADE_BLACK = 3, //!< fade in fade out (black)
68 ARM_2D_SCENE_SWITCH_MODE_SLIDE_LEFT = 4, //!< slide left
69 ARM_2D_SCENE_SWITCH_MODE_SLIDE_RIGHT, //!< slide right
70 ARM_2D_SCENE_SWITCH_MODE_SLIDE_UP, //!< slide up
71 ARM_2D_SCENE_SWITCH_MODE_SLIDE_DOWN, //!< slide down
72 ARM_2D_SCENE_SWITCH_MODE_ERASE_LEFT = 8, //!< erase to the right
73 ARM_2D_SCENE_SWITCH_MODE_ERASE_RIGHT, //!< erase to the left
74 ARM_2D_SCENE_SWITCH_MODE_ERASE_UP, //!< erase to the top
75 ARM_2D_SCENE_SWITCH_MODE_ERASE_DOWN, //!< erase to the bottom
76
77 /* valid switching visual effects end */
78 __ARM_2D_SCENE_SWITCH_MODE_VALID, //!< For internal user only
79
80
81
82 ARM_2D_SCENE_SWITCH_MODE_IGNORE_OLD_BG = _BV(8), //!< ignore the background of the old scene
83 ARM_2D_SCENE_SWITCH_MODE_IGNORE_OLD_SCEBE = _BV(9), //!< ignore the old scene
84 ARM_2D_SCENE_SWITCH_MODE_IGNORE_NEW_BG = _BV(10), //!< ignore the background of the new scene
85 ARM_2D_SCENE_SWITCH_MODE_IGNORE_NEW_SCEBE = _BV(11), //!< ignore the new scene
86
87 ARM_2D_SCENE_SWITCH_MODE_DEFAULT_BG_WHITE = 0 << 12, //!< use white as default background
88 ARM_2D_SCENE_SWITCH_MODE_DEFAULT_BG_BLACK = 1 << 12, //!< use black as default background
89 ARM_2D_SCENE_SWITCH_MODE_DEFAULT_BG_USER = 2 << 12, //!< use user defined default background
90
91 __ARM_2D_SCENE_SWTICH_MODE_IGNORE_msk = 0x0F << 8, //!< For internal user only
92 __ARM_2D_SCENE_SWTICH_MODE_IGNORE_pos = 8, //!< For internal user only
93 __ARM_2D_SCENE_SWTICH_MODE_DEFAULT_BG_msk = 3 << 12, //!< For internal user only
94 __ARM_2D_SCENE_SWTICH_MODE_DEFAULT_BG_pos = 12, //!< For internal user only
95} arm_2d_scene_player_switch_mode_t;
96
97typedef union __arm_2d_helper_scene_switch_t {
98 struct {
99 uint8_t chMode; //!< the switch visual effect
100 uint8_t bIgnoreOldSceneBG : 1; //!< when set, ignore the background of the old scene
101 uint8_t bIgnoreOldScene : 1; //!< when set, ignore the old scene
102 uint8_t bIgnoreNewSceneBG : 1; //!< when set, ignore the background of the new scene
103 uint8_t bIgnoreNewScene : 1; //!< when set, ignore the new scene
104 uint8_t u2DefaultBG : 2; //!< the default background
105 uint8_t : 2;
106 } Feature;
107 uint16_t hwSetting; //!< the setting value
108}__arm_2d_helper_scene_switch_t;
109
110/*!
111 * \brief a class for describing scenes which are the combination of a
112 * background and a foreground with a dirty-region-list support
113 *
114 */
115typedef struct arm_2d_scene_t arm_2d_scene_t;
116struct arm_2d_scene_t {
117 arm_2d_scene_t *ptNext; //!< next scene
118 arm_2d_region_list_item_t *ptDirtyRegion; //!< dirty region list for the foreground
119 arm_2d_helper_draw_handler_t *fnBackground; //!< the function pointer for the background
120 arm_2d_helper_draw_handler_t *fnScene; //!< the function pointer for the foreground
121 void (*fnOnBGStart)(arm_2d_scene_t *ptThis); //!< on-start-drawing-background event handler
122 void (*fnOnBGComplete)(arm_2d_scene_t *ptThis); //!< on-complete-drawing-background event handler
123 void (*fnOnFrameStart)(arm_2d_scene_t *ptThis); //!< on-frame-start event handler
124 void (*fnOnFrameCPL)(arm_2d_scene_t *ptThis); //!< on-frame-complete event handler
125
126 /*!
127 * \note We use fnDepose to free the resources
128 */
129 void (*fnDepose)(arm_2d_scene_t *ptThis); //!< on-scene-depose event handler
130 struct {
131 uint8_t bOnSwitchingIgnoreBG : 1; //!< ignore background during switching period
132 uint8_t bOnSwitchingIgnoreScene : 1; //!< ignore forground during switching period
133 };
134};

136/*!
137 * \brief a class to manage scenes
138 *
139 */
140typedef struct arm_2d_scene_player_t {
141 inherit(arm_2d_helper_pfb_t); //!< inherit from arm_2d_helper_pfb_t
142
143 ARM_PRIVATE(
144 struct {
145 arm_2d_scene_t *ptHead; //!< points to the head of the FIFO
146 arm_2d_scene_t *ptTail; //!< points to the tail of the FIFO
147 } SceneFIFO; //!< Scene FIFO
148
149 struct {
150 uint8_t bNextSceneReq : 1; //!< a flag to request switching-to-the next-scene
151 uint8_t bSwitchCPL : 1; //!< indication of scene switching completion
152 uint8_t : 6;
153 uint8_t chState; //!< the state of the FSM used by runtime.
154 } Runtime; //!< scene player runtime
155
156 struct {
157 union {
158 uint8_t chState;
159 struct {
160 uint8_t chState;
161 uint8_t chOpacity;
162 bool bIsFadeBlack;
163 }Fade;
164 };
165 __arm_2d_helper_scene_switch_t tConfig; //!< the switching configuration
166 uint16_t hwPeriod; //!< the switching should finish in specified millisecond
167 int64_t lTimeStamp;
168 }Switch;
169 )
170} arm_2d_scene_player_t;

243/*!
244 * \brief the scene player task function
245 *
246 * \param[in] ptThis the target scene player
247 *
248 * \note the event sequence of a scene:
249 * 1. when fnBackground is valid
250 * - invoke fnOnBGStart when it is valid
251 * - invoke fnBackground
252 * - invoke fnOnBGComplete when it is valid
253 * 2. invoke fnOnFrameStart when it is valid
254 * 3. invoke fnScene
255 * 4. invoke fnOnFrameCPL when it is valid
256 * 5. Check bNextSceneReq
257 * - false (0), go back to step 2
258 * - true, invoke fnDepose when it is valid and switch to the next scene
259 *
260 */
261extern
262ARM_NONNULL(1)
263arm_fsm_rt_t arm_2d_scene_player_task(arm_2d_scene_player_t *ptThis);
