58/*!
59 * \brief a class for describing scenes which are the combination of a
60 * background and a foreground with a dirty-region-list support
61 *
62 */
63typedef struct arm_2d_scene_t arm_2d_scene_t;
64struct arm_2d_scene_t {
65 arm_2d_scene_t *ptNext; //!< next scene
66 arm_2d_region_list_item_t *ptDirtyRegion; //!< dirty region list for the foreground
67 arm_2d_helper_draw_handler_t *fnBackground; //!< the function pointer for the background
68 arm_2d_helper_draw_handler_t *fnScene; //!< the function pointer for the foreground
69 void (*fnOnBGStart)(arm_2d_scene_t *ptThis); //!< on-start-drawing-background event handler
70 void (*fnOnBGComplete)(arm_2d_scene_t *ptThis); //!< on-complete-drawing-background event handler
71 void (*fnOnFrameStart)(arm_2d_scene_t *ptThis); //!< on-frame-start event handler
72 void (*fnOnFrameCPL)(arm_2d_scene_t *ptThis); //!< on-frame-complete event handler
73 void (*fnDepose)(arm_2d_scene_t *ptThis); //!< on-scene-depose event handler
74 /*!
75 * \note We use fnDepose to free the resources
76 */
77};
78
79/*!
80 * \brief a class to manage scenes
81 *
82 */
83typedef struct arm_2d_scene_player_t {
84 inherit(arm_2d_helper_pfb_t); //!< inherit from arm_2d_helper_pfb_t
85
86 ARM_PRIVATE(
87 struct {
88 arm_2d_scene_t *ptHead; //!< points to the head of the FIFO
89 arm_2d_scene_t *ptTail; //!< points to the tail of the FIFO
90 } SceneFIFO; //!< Scene FIFO
91
92 struct {
93 uint8_t bNextSceneReq : 1; //!< a flag to request switching-to-the next-scene
94 uint8_t : 7;
95 uint8_t chState; //!< the state of the FSM used by runtime.
96 } Runtime; //!< scene player runtime
97 )
98} arm_2d_scene_player_t;

104/*!
105 * \brief register a list of scenes to a scene player
106 *
107 * \param[in] ptThis the target scene player
108 * \param[in] ptSceneList the first node of a scene list
109 *
110 * \note When you register new scene array to a scene player, existing scenes
111 * (if any) will be deposed one by one by the scene player first. This is
112 * different from appending scene(s), as this operation is replacing the
113 * old (existing) scene with new ones.
114 */
115extern
116ARM_NONNULL(1)
117void arm_2d_user_scene_player_set_scenes( arm_2d_scene_player_t *ptThis,
118 arm_2d_scene_t *ptSceneList);
119
120/*!
121 * \brief append a list of scenes to a scene player
122 *
123 * \param[in] ptThis the target scene player
124 * \param[in] ptSceneList the first node of a scene list
125 */
126extern
127ARM_NONNULL(1)
128void arm_2d_user_scene_player_append_scenes(arm_2d_scene_player_t *ptThis,
129 arm_2d_scene_t *ptSceneList);
130
131/*!
132 * \brief request switching to the next scene safely
133 *
134 * \param[in] ptThis the target scene player
135 *
136 * \note Once received a request, the scene player will only switch to the
137 * next scene at the end of a frame.
138 */
139extern
140ARM_NONNULL(1)
141void arm_2d_user_scene_player_next_scene(arm_2d_scene_player_t *ptThis);
142
143/*!
144 * \brief the scene player task function
145 *
146 * \param[in] ptThis the target scene player
147 *
148 * \note the event sequence of a scene:
149 * 1. when fnBackground is not NULL
150 * - invoke fnOnBGStart when it is valid
151 * - invoke fnBackground
152 * - invoke fnOnBGComplete when it is valid
153 * 2. invoke fnOnFrameStart when it is valid
154 * 3. invoke fnScene
155 * 4. invoke fnOnFrameCPL when it is valid
156 * 5. Check bNextSceneReq
157 * - false (0), go back to step 2
158 * - true, invoke fnDepose when it is valid and switch to the next scene
159 *
160 */
161extern
162ARM_NONNULL(1)
163arm_fsm_rt_t arm_2d_user_scene_player_task(arm_2d_scene_player_t *ptThis);
