78/*! \brief 3x16-bit packed RGB color
79 * autovectorizer friendly format
80 */
81typedef union {
82 uint16_t BGRA[4];
83 struct {
84 uint16_t B;
85 uint16_t G;
86 uint16_t R;
87 uint16_t A;
88 };
89} __arm_2d_color_fast_rgb_t;

94/*----------------------------------------------------------------------------*
95 * RGB565 channels extraction/packing *
96 *----------------------------------------------------------------------------*/
97
98/*!
99 * \brief unpack a rgb565 colour into a given __arm_2d_color_fast_rgb_t object
100 * \param[in] hwColour the target rgb565 colour
101 * \param[out] ptRGB a __arm_2d_color_fast_rgb_t object
102 */
103ARM_NONNULL(2)
104__STATIC_INLINE void __arm_2d_rgb565_unpack(uint16_t hwColor,
105 __arm_2d_color_fast_rgb_t * ptRGB)
106{
107 assert(NULL != ptRGB);
108
109 /* uses explicit extraction, leading to a more efficient autovectorized code */
110 uint16_t maskRunpk = 0x001f, maskGunpk = 0x003f;
111
112 ptRGB->B = (uint16_t) ((hwColor & maskRunpk) << 3);
113 ptRGB->R = (uint16_t) ((hwColor >> 11) << 3);
114 ptRGB->G = (uint16_t) (((hwColor >> 5) & maskGunpk) << 2);
115
116 ptRGB->A = 0xFF;
117}
118
119/*!
120 * \brief generate a rgb565 colour from a __arm_2d_color_fast_rgb_t object
121 * \param[in] ptRGB the target __arm_2d_color_fast_rgb_t object
122 * \return uint16_t a rgb565 colour
123 */
124ARM_NONNULL(1)
125__STATIC_INLINE uint16_t __arm_2d_rgb565_pack(__arm_2d_color_fast_rgb_t * ptRGB)
126{
127 assert(NULL != ptRGB);
128
129 arm_2d_color_rgb565_t tOutput = {
130 .u5R = (uint16_t) ptRGB->R >> 3,
131 .u6G = (uint16_t) ptRGB->G >> 2,
132 .u5B = (uint16_t) ptRGB->B >> 3,
133 };
134 return tOutput.tValue;
135}
136
137/*!
138 * \brief generate a cccn888 colour from a __arm_2d_color_fast_rgb_t object
139 * \param[in] ptRGB the target __arm_2d_color_fast_rgb_t object
140 * \return uint32_t a cccn888 colour
141 * \note the alpha channel will be kept in the output value
142 */
143ARM_NONNULL(1)
144__STATIC_INLINE uint32_t __arm_2d_cccn888_pack(__arm_2d_color_fast_rgb_t * ptRGB)
145{
146 assert(NULL != ptRGB);
147
148 arm_2d_color_bgra8888_t tOutput = {
149 .u8R = (uint16_t) ptRGB->R,
150 .u8G = (uint16_t) ptRGB->G,
151 .u8B = (uint16_t) ptRGB->B,
152 .u8A = (uint16_t) ptRGB->A,
153 };
154 return tOutput.tValue;
155}
