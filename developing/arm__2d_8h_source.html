162/*!
163 * \brief wait asynchronous operation complete
164 * \param[in] ptOP the address of the target OP (NULL means using the default OP)
165 * \retval true sync up with operation
166 * \retval false operation is busy
167 */
168extern
169bool arm_2d_op_wait_async(arm_2d_op_core_t *ptOP);
170
171/*!
172 \brief get the status of a specified OP,
173 \details usually, it is used after calling arm_2d_op_wait_async().
174 E.g.
175 \code
176 //! wait for previous operation complete
177 do {
178 arm_2d_op_wait_async();
179 arm_2d_op_status_t tStatus = arm_2d_get_op_status();
180 if (tStatus.bIOError) {
181 //! error detected
182 ...
183 } else if (tStatus.bOpCpl) {
184 break;
185 }
186 } while(true);
187 \endcode
188 \param ptOP the address of the target OP (NULL means using the default OP)
189 \return arm_2d_op_status_t the operation status
190 */
191extern
192arm_2d_op_status_t arm_2d_get_op_status(arm_2d_op_core_t *ptOP);
193
194/*!
195 * \brief arm-2d pixel pipeline task entery
196 * \note This function is *TRHEAD-SAFE*
197 * \param ptTask the address of an arm-2d task control block
198 * \retval arm_fsm_rt_cpl The sub-task FIFO is empty, the caller can wait for a
199 * semaphore set by arm_2d_notif_sub_task_fifo_task_arrive()
200 * \retval arm_fsm_rt_on_going The arm_2d_task yields
201 * \retval arm_fsm_rt_async You shouldn't see this value
202 * \retval arm_fsm_rt_wait_for_obj hardware accelerator wants to sync-up with applications.
203 * \retval (<0) Serious error is detected.
204 */
205extern
206arm_fsm_rt_t arm_2d_task(arm_2d_task_t *ptTask);
